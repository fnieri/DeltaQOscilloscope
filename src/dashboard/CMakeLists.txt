        Dashboard.cpp
        Dashboard.h
        Plot.cpp
        Plot.h
        DeltaQHelpers.h
        DeltaQHelpers.cpp
)

        PUBLIC
        ${CMAKE_SOURCE_DIR}

        main.cpp
)

        PRIVATE
        ${PREFIX}_dashboard
)
